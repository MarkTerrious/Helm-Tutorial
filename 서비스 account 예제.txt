# 1. Prometheus용 ServiceAccount
apiVersion: v1
kind: ServiceAccount
metadata:
  name: prometheus
  namespace: monitoring

---
# 2. Prometheus가 필요로 하는 권한
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRole
metadata:
  name: prometheus
rules:
- apiGroups: [""]
  resources: ["nodes", "services", "endpoints", "pods"]
  verbs: ["get", "list", "watch"]
- apiGroups: ["extensions"]
  resources: ["ingresses"]
  verbs: ["get", "list", "watch"]
- apiGroups: ["networking.k8s.io"]
  resources: ["ingresses"]
  verbs: ["get", "list", "watch"]

---
# 3. ClusterRoleBinding
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRoleBinding
metadata:
  name: prometheus
subjects:
- kind: ServiceAccount
  name: prometheus
  namespace: monitoring
roleRef:
  kind: ClusterRole
  name: prometheus
  apiGroup: rbac.authorization.k8s.io

---
# 4. Deployment에서 지정!
apiVersion: apps/v1
kind: Deployment
metadata:
  name: prometheus
  namespace: monitoring
spec:
  template:
    spec:
      serviceAccountName: prometheus  # ← 여기가 핵심!
      containers:
      - name: prometheus
        image: prom/prometheus


subjects:
- kind: ServiceAccount          # ← 이 권한을 누가 받을까?
  name: prometheus              # ← ServiceAccount 이름
  namespace: monitoring         # ← 그 ServiceAccount가 있는 네임스페이스

roleRef:
  kind: ClusterRole             # ← 어떤 권한 뭉치를 줄까?
  name: prometheus              # ← ClusterRole 이름
  apiGroup: rbac.authorization.k8s.io   # ← RBAC 리소스는 이 apiGroup에 속해 있음
